---
title: "omitKeys"
description: "Omit certain keys from an object."
---

#### Install

```bash
npx shadcn@latest add https://fncn.mvze.net/r/omit-keys.json
```

#### Usage

```ts
import { capitalize } from "@/lib/omit-keys";

const users = {
  name: "John Doe",
  email: "john@doe.com",
  password: "supersecret",
};

const safe = omitKeys(users, ["password"]);

console.log(safe.password); // => undefined
```
